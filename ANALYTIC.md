Gasto Operacional / Margen Financiero Neto

```
!GASTO OPERTACIONAL/MARGEN FINANCIERO NETO =
VAR INTERESES_GANADOS=CALCULATE(SUM('1-CONSOLIDADO-'[51]))
VAR INTERESES_CAUSADOS=CALCULATE(SUM('1-CONSOLIDADO-'[41]))
VAR MARGEN_NETO_INTRESES= (INTERESES_GANADOS-INTERESES_CAUSADOS)
VAR COMISIONES_GANADAS=CALCULATE(SUM('1-CONSOLIDADO-'[52]))
VAR INGRE_SERVICIO=CALCULATE(SUM('1-CONSOLIDADO-'[54]))
VAR COMISI_CAUSADAS=CALCULATE(SUM('1-CONSOLIDADO-'[42]))
VAR UTILIDADES_FINANCIERAS=CALCULATE(SUM('1-CONSOLIDADO-'[53]))  
VAR PERDIDAS_FINANCIERAS=CALCULATE(SUM('1-CONSOLIDADO-'[43]))  
VAR MARGEN_BRUTO_FINANCIERO=(MARGEN_NETO_INTRESES+COMISIONES_GANADAS+INGRE_SERVICIO-COMISI_CAUSADAS+UTILIDADES_FINANCIERAS-PERDIDAS_FINANCIERAS)
VAR PROVISIONES=CALCULATE(SUM('1-CONSOLIDADO-'[44]))
VAR MARGEN_FINANCIERO_NETO=MARGEN_BRUTO_FINANCIERO-PROVISIONES
VAR GASTO_OPERATIVO=CALCULATE(SUM('1-CONSOLIDADO-'[45]))
VAR GAST_OPERA_MRG_FINAN_NETO=GASTO_OPERATIVO/MARGEN_FINANCIERO_NETO
RETURN IFERROR(GAST_OPERA_MRG_FINAN_NETO,0)
```

```
!GASTO OPERA ESTIMADO =
VAR GAST_OPE=SUMX('1-CONSOLIDADO-','1-CONSOLIDADO-'[45]/'1-CONSOLIDADO-'[MES]*12)
RETURN GAST_OPE

!CARTERA BRUTA =
var A=CALCULATE(sum('1-CONSOLIDADO-'[14]))
var B=CALCULATE(sum('1-CONSOLIDADO-'[1499]))
RETURN A-B

!GASTO OPERATIVO ESTIMADO/CARTERA BRUTA =
VAR A= SUM('1-CONSOLIDADO-'[45])
VAR RESULT=[!GASTO OPERA ESTIMADO]/[!CARTERA BRUTA]

COSTO PROMEDIO DE FONDEO

```

!OBLIGACIO FINANC =
var A=CALCULATE(sum('1-CONSOLIDADO-'[26]))
RETURN A

!OBLIGACIO PUBLI =
var A=CALCULATE(sum('1-CONSOLIDADO-'[21]))
RETURN A

!COSTO PROMEDIO FONDEO =
VAR MES=VALUES(CALENDARIO[MES])
VAR A=AVERAGEX(DATESINPERIOD(CALENDARIO[Date],MAX(CALENDARIO[Date]),-MES,MONTH),'1-CONSOLIDADO-'[!OBLIGACIO FINANC])
VAR B=AVERAGEX(DATESINPERIOD(CALENDARIO[Date],MAX(CALENDARIO[Date]),-MES,MONTH),'1-CONSOLIDADO-'[!OBLIGACIO PUBLI])
VAR C=CALCULATE(SUM('1-CONSOLIDADO-'[41]))
VAR PART=C/(A+B)/[#MES]*12
RETURN PART
